? Java Code — Find Cousins of a Node in a Binary Tree

class Solution {

    static class TreeNode {
        int val;
        TreeNode left, right;
        TreeNode(int val) { this.val = val; }
    }

    int targetLevel = -1;
    TreeNode targetParent = null;
    List<Integer> result = new ArrayList<>();

    public List<Integer> findCousins(TreeNode root, int target) {
        dfs(root, null, 0, target);
        return result;
    }

    private void dfs(TreeNode node, TreeNode parent, int level, int target) {
        if (node == null) return;

        // Step 1: Found target ? store its parent + level
        if (node.val == target) {
            targetLevel = level;
            targetParent = parent;
        }

        // Step 2: If same level & not sibling ? add to result
        if (level == targetLevel && parent != targetParent && node.val != target) {
            result.add(node.val);
        }

        // Step 3: Continue DFS
        dfs(node.left, node, level + 1, target);
        dfs(node.right, node, level + 1, target);
    }
}
